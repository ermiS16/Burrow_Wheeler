MenuBar:
    menu_bar_width
    menu_bar_height
    
ControlPanel:
    control_panel_width = window.width
    control_panel_height = window_height*0.1 (10% von Parent)
    control_panel_x = 0
    control_panel_y = menu_bar_height
    
    x_start = 0
    y_start = control_panel_y
    
    elem_margin_x = control_panel_width * 0.02 (2% von Parent)
    elem_margin_y = control_panel_height * 0.01 (1% von Parent)
    
    control_panel_padding_top = control_panel_height * 0.02 (2% von Parent)
    
    input_field_label_width
    input_field_width
    delta_field_label_width
    delta_field_width
    transform_btn_width
    next_btn_width
    prev_btn_width
    slider_width
    direction_box_width
    
    COUNT_ELEM_X = 7
    elem_all_width = (elem_margin_x*COUNT_ELEM_X) 
                    + input_field_label_width 
                    + input_field_width 
                    + transform_btn_width 
                    + next_btn_width
                    + prev_btn_width
                    + slider_width
                    + direction_box_width
                    
    elem_all_width_half = elem_all_width/2    
    control_panel_center_x = control_panel_width / 2
    
    x_start = control_panel_center_x - elem_all_width_half
    y_start = control_panel_y + control_panel_padding_top
    input_field_label.move(x_start, y_start)
    
    x_start = x_start + input_field_label_width + elem_margin_x
    input_field.move(x_start, y_start)
    
    x_start = x_start + direction_box_width + elem_margin_x
    direction_box.move(x_start, y_start)
    
    x_start = x_start + next_btn_width + elem_margin_x
    next_btn.move(x_start, y_start)

    x_start = x_start + prev_btn_width + elem_margin_x
    prev_btn.move(x_start, y_start)
    
    x_start = x_start + slider_width + elem_margin_x
    slider.move(x_start, y_start)
    
    x_start control_panel_center_x - elem_all_width_half
    y_start = y_start + elem_margin_y
    delta_field_label.move(x_start, y_start)
    
    x_start = x_start + delta_field_width + elem_margin_x
    delta_field.move(x_start, y_start)
    
    x_start = x_start + transform_btn_width + elem_margin_x
    transform_btn.move(x_start, y_start)
    
    
Content:
    content_width = window.width
    content_height = window_height - menu_bar_height + control_panel_height
    content_x = 0
    content_y = menu_bar_height + control_panel_height
